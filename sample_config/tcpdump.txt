$ tcpdump -i eth0 -G 86400 -Z root -w test6s_%m%d.pcap host 133.14.14.246 and port 443 &
$ tcpdump -i eth0 -G 86400 -Z root -w test7s_%m%d.pcap host 133.14.14.247 and port 443 &
$ tcpdump -i eth0 -G 86400 -Z root -w test7_%m%d.pcap host 133.14.14.247 and port 80 &
$ tcpdump -i eth0 -G 86400 -Z root -w test8s_%m%d.pcap host 133.14.14.248 and port 443 &
$ tcpdump -i eth0 -G 86400 -Z root -w test8_%m%d.pcap host 133.14.14.248 and port 80 &

#　host と port　で 133.14.14.24*: 443 or 80　へのパケットと、からのパケットを指定
#　-i　でインターフェイス指定　//これなしだと、一番値の低いものに自動指定される
#　-G　でファイルを１日ごとに作成することを指定
#　-Z　でコマンド実行ユーザの指定と & でバックグラウンドでの実行を指定
#　-w　で書き込みファイルの指定
#　これをハニポのログ収集時と同時に開始させる -> 06:00
